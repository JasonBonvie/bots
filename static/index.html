Signal Computation (bot/signals.py) 
def compute_signals(symbol: str = "BTC/USDT", bars: int = 50) -> SignalResult:
    """
    Compute momentum signals for the given symbol.

    Scoring system (0 to 4.5 for each direction):
    - +1 if EMA8 > EMA21 → score_up, else score_dn
    - +1 if 50 < RSI < 70 → score_up; if 30 < RSI < 50 → score_dn
    - +1 if MACD > Signal → score_up, else score_dn
    - +0.5 if volume > 1.3x avg (both directions)
    - +1 if close_pos > 0.65 → score_up; if close_pos < 0.35 → score_dn
    """

    # Fetch 15m OHLCV from Binance (fallback to Coinbase)
    exchange = create_exchange("binance")
    df = fetch_ohlcv_with_retry(exchange, symbol, "15m", bars)

    # Compute indicators with pandas-ta
    df["rsi"] = ta.rsi(df["close"], length=14)
    df["ema8"] = ta.ema(df["close"], length=8)
    df["ema21"] = ta.ema(df["close"], length=21)

    macd_df = ta.macd(df["close"], fast=12, slow=26, signal=9)
    df["macd"] = macd_df["MACD_12_26_9"]
    df["macd_signal"] = macd_df["MACDs_12_26_9"]

    # Volume ratio: current / 20-bar SMA
    df["vol_sma20"] = df["volume"].rolling(window=20).mean()
    df["volume_ratio"] = df["volume"] / df["vol_sma20"]

    # Close position in candle range
    df["close_position"] = (df["close"] - df["low"]) / (df["high"] - df["low"])

    # Score calculation
    score_up, score_dn = 0.0, 0.0

    if ema8 > ema21: score_up += 1
    else: score_dn += 1

    if 50 < rsi < 70: score_up += 1
    elif 30 < rsi < 50: score_dn += 1

    if macd > macd_signal: score_up += 1
    else: score_dn += 1

    if volume_ratio > 1.3:
        score_up += 0.5
        score_dn += 0.5

    if close_pos > 0.65: score_up += 1
    elif close_pos < 0.35: score_dn += 1

    # Direction determination
    if score_up >= 3.5: direction = "UP"
    elif score_dn >= 3.5: direction = "DOWN"
    else: direction = "SKIP"

    return SignalResult(...)

• Rate limiting handled with exponential backoff (2s base delay, 3 retries)
• Binance primary, Coinbase fallback for OHLCV data
• WebSocket broadcasts at :00, :15, :30, :45 + 2s buffer for candle close
• All times synced to UTC, displayed in local timezone
